    # CMS 로그인 연동 가이드

    SSO(Single Sign On) 및 외부데이터를 통한 로그인 처리 후 CMS 로그인 세션 처리

    # 로그인 처리 후 리턴 페이지 추가 로직

    TIP

    기본적인 연동정보만 기술되어 있고 그룹코드,직급코드,회원유형등의 추가 적인 정보까지 연동하기 위해선 추가 로직 작성이 필요합니다.

    ...

  request.getSession().invalidate();
  HttpSession session = request.getSession( true ); // 새로운 세션을 생성

  //-------------------------------------
  // CMS LoginVO 생성
  //-------------------------------------
  LoginVO loginVO = new LoginVO();
  
  loginVO.setMemberId(memberId);   //  
  loginVO.setMemberNm(memberNm);   // 
  loginVO.setEmail(email);    // 이메일 사용하면 작성
  loginVO.setMemberMobile(memberMobile); // 핸드폰번호 사용하면 작성
  loginVO.setMemberTel(memberTel);   // 전화번호 사용하면  작성
  loginVO.setCurrentLoginType( LoginType.ID )  ;  // 필수
  loginVO.setIpAddress(CMS.getRemoteAddr(request)); // 필수
  loginVO.setLoginTime( new Date() );   // 필수
  loginVO.addLoginType( LoginType.ID ); // 필수

  
  SessionMgr.setLoginVO(loginVO);
